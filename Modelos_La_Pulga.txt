Modelo Relacional:

Usuario (id PK, password, tipo, nombre, activo, token)
Cliente (id PK, nombre, ap, am, correo, ciudad, col, calle, noint, noext, cp, foto, fec_registro, id_usu FK)
Administrador (id PK, nombre, ap, am, correo, foto, fec_registro, id_usu FK)
Categoria (id PK, categoria)
Producto (id PK, nombre, prec, foto, estado, cantidad, id_cat FK)
Venta (id PK, fech, folio, total, id_cli FK, envio)
Detalle_Venta (id PK, subtotal, cant, prec, id_vent FK, id_pro FK)
Telefono (id PK, telefono, id_cli FK)
Lista (id PK, id_pro FK, id_cli FK)
Carrusel (id PK, foto)
